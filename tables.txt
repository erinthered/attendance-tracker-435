CREATE TABLE Users (
    user_id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(50) NOT NULL,
    email VARCHAR(100) NOT NULL,
    password_hash VARCHAR(150) NOT NULL,
    user_type ENUM('student', 'teacher') NOT NULL,
    PRIMARY KEY(user_id)
);

CREATE TABLE Classes (
	class_id INT NOT NULL AUTO_INCREMENT,
	name VARCHAR(50) NOT NULL,
	section VARCHAR(50) NOT NULL,
    attendance_code VARCHAR(50),
    enrollment_code VARCHAR(50) NOT NULL,
    professor_id INT NOT NULL,
	PRIMARY KEY(class_id),
	FOREIGN KEY(professor_id) REFERENCES Users(user_id) ON DELETE CASCADE
);

CREATE TABLE Enrollment (
    class_id INT,
    user_id INT,
    dropped BOOLEAN DEFAULT false,
    PRIMARY KEY(class_id, user_id),
    FOREIGN KEY(class_id) REFERENCES Classes(class_id) ON DELETE CASCADE,
    FOREIGN KEY(user_id) REFERENCES Users(user_id)
);

CREATE TABLE Attendance (
  class_id INT,
  user_id INT,
  date DATE,
  PRIMARY KEY(class_id, user_id),
  FOREIGN KEY(class_id) REFERENCES Enrollment(class_id) ON DELETE CASCADE,
  FOREIGN KEY(user_id) REFERENCES Enrollment(user_id)
);
